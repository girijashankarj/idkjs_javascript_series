Hoisting Quirk

JavaScript’s hoisting behavior can be perplexing. It allows variable and function declarations to be moved to the top of their containing scope during compilation. Understanding this quirk is crucial for writing clean and predictable code.

// Example 1: Variable Hoisting
console.log(x); // undefined
var x = 5; 

// Example 2: Function Hoisting
greet(); // Hello!
function greet() {
  console.log("Hello!");
}

Type Coercion Quirk

JavaScript performs type coercion when comparing values of different types. This can lead to unexpected results. Understanding how type coercion works is essential for avoiding subtle bugs.

// Example 1: Type Coercion
let number = 5;
let string = "10";

let result = number + string;

console.log(result); // Output: "510"

// Example 2: Numeric String to Number
let numericString = "42";
let number = 10;

let result = numericString + number;

console.log(result); // Output: "4210"

// Example 3: Boolean to Number
let booleanValue = true;

let numberResult = 10 + booleanValue;
console.log(numberResult); // Output: 11

let stringResult = "Hello, " + booleanValue;
console.log(stringResult); // Output: "Hello, true"

// Example 4: Undefined and Null Coercion
let undefinedValue;
let nullValue = null;

console.log(undefinedValue + 10); // Output: NaN (Not-a-Number)
console.log(nullValue + "Hello"); // Output: "nullHello"

// Example 5: Truthy and Falsy Values
let truthyValue = "Hello";
let falsyValue = "";

console.log(truthyValue == true); // Output: false
console.log(falsyValue == false); // Output: true

The this Keyword Quirk

The behavior of the this keyword can be confusing, especially in different contexts such as functions, objects, and event handlers. Knowing when and how to use it correctly is vital.

// Example: Understanding 'this' in Different Contexts
// Global Context
console.log(this === window); // In a browser environment, 'this' refers to the global object, which is 'window'.

function globalFunction() {
  console.log(this === window); // 'this' inside a regular function still refers to the global object.
}
globalFunction();

// Object Method Context
const myObject = {
  property: 'I am a property',
  method: function () {
    console.log(this.property); // 'this' refers to the object (myObject) in which the method is called.
  },
};

myObject.method(); // Outputs: 'I am a property'

// Constructor Function Context
function Person(name) {
  this.name = name;
  console.log(this.name); // 'this' refers to the newly created instance when used with a constructor function.
}

const person1 = new Person('Alice'); // Outputs: 'Alice'

// Event Handler Context (DOM)
const button = document.querySelector('button');

button.addEventListener('click', function () {
  console.log(this); // 'this' refers to the element that triggered the event (the button).
});

Closures Quirk

Closures can lead to unexpected memory retention and memory leaks if not used carefully. Understanding closure behavior is crucial for efficient memory management.

// Example: Closure and Memory Retention
function createCounter() {
  let count = 0;
  return function () {
    return ++count;
  };
}
const counter1 = createCounter();
const counter2 = createCounter();
console.log(counter1()); // 1
console.log(counter1()); // 2
console.log(counter2()); // 1

Event Loop Quirk

JavaScript’s event loop and asynchronous nature can be challenging to grasp fully. It’s essential to understand how the event loop works to write efficient, non-blocking code.

// Example: Asynchronous Code with Callbacks
// Simulating an asynchronous operation with setTimeout
function simulateAsyncOperation(callback) {
  setTimeout(function () {
    callback("Operation completed");
  }, 2000);
}

// Callback function to handle the result
function handleResult(message) {
  console.log("Result:", message);
}

console.log("Start of the operation");
simulateAsyncOperation(handleResult);
console.log("End of the operation");

// Output:
// Start of the operation
// End of the operation
// (after 2 seconds) Result: Operation completed

Promises and Async/Await Quirk

Handling asynchronous operations using Promises and Async/Await can be intricate. Mastery of these concepts is essential for modern web development.

// Example: Promises Quirks

function delay(ms) {
  return new Promise((resolve) => {
    setTimeout(resolve, ms);
  });
}

function throwError() {
  return new Promise((resolve, reject) => {
    reject(new Error("Something went wrong"));
  });
}

async function unhandledRejection() {
  throw new Error("Unhandled Promise Rejection");
}

const promises = [
  delay(1000).then(() => {
    console.log("First .then");
    return delay(1000);
  }),
  delay(1000).then(() => {
    console.log("Second .then");
    return delay(1000);
  }),
  delay(1000).then(() => {
    console.log("Third .then");
  }),
];

console.log("Start of script");

throwError()
  .then(() => {
    console.log("Success");
  })
  .catch((error) => {
    console.error("Error:", error.message);
  });

unhandledRejection();

Promise.all(promises)
  .then((results) => {
    console.log("Results:", results);
  })
  .catch((error) => {
    console.error("Error:", error.message);
  });

// Expected Output:
// Start of script
// First .then (after 1 second)
// Second .then (after 2 seconds)
// Third .then (after 3 seconds)
// Error: Something went wrong
// Error: Unhandled Promise Rejection
// Results: [undefined, undefined, undefined] (after 3 seconds)

Promises and Async/Await Quirk

Handling asynchronous operations using Promises and Async/Await can be intricate. Mastery of these concepts is essential for modern web development.

// Example: Async/Await Quirk with Promises

function resolveAfter2Seconds() {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve("Resolved after 2 seconds");
    }, 2000);
  });
}

async function asyncFunction() {
  console.log("Start of asyncFunction");
  const result = await resolveAfter2Seconds();
  console.log("Inside asyncFunction:", result);
  console.log("End of asyncFunction");
}

console.log("Before calling asyncFunction");
asyncFunction();
console.log("After calling asyncFunction");

// Expected Output:
// Before calling asyncFunction
// Start of asyncFunction
// After calling asyncFunction
// (after 2 seconds) Inside asyncFunction: Resolved after 2 seconds
// End of asyncFunction

Prototypal Inheritance Quirk

JavaScript uses prototypal inheritance, which can differ from classical inheritance found in other languages. Understanding how prototypes work is fundamental for object-oriented programming in JavaScript.

// Example: Prototypal Inheritance Quirk

// Parent "Person" constructor function
function Person(name) {
  this.name = name;
}

Person.prototype.greet = function () {
  console.log(`Hello, my name is ${this.name}`);
};

// Child "Student" constructor function
function Student(name, grade) {
  Person.call(this, name); // Call the parent constructor to set the name
  this.grade = grade;
}

// Inherit from Person's prototype
Student.prototype = Object.create(Person.prototype);
Student.prototype.constructor = Student; // Reset the constructor

// Add a method to the Student prototype
Student.prototype.study = function () {
  console.log(`${this.name} is studying.`);
};

// Create instances
const person1 = new Person("Alice");
const student1 = new Student("Bob", "A");

person1.greet(); // Output: Hello, my name is Alice
student1.greet(); // Output: Hello, my name is Bob
student1.study(); // Output: Bob is studying.

// Quirk: Modifying a prototype property affects all instances
Student.prototype.greet = function () {
  console.log(`Hey, I'm ${this.name}`);
};

person1.greet(); // Output: Hey, I'm Alice
student1.greet(); // Output: Hey, I'm Bob
